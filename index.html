        'name' : {
            'type' : 'string'

    }
}

Validation

{
    'type' : 'object',
    'properties': {
        'name' : {
              'type':'string'
              'pattern':'[a-zA-Z]*'
             },
        'email' :  {
              'type':'string'
              'format':'email'
              },
    },
    'required' : ['name']
}

Recursive data structures


{
    'type' : 'object',
    'properties': {
        'name' : {'type': 'string'}
        'friend' : {'$ref': '#'}
    }
}
          

{
    'name': "Willi",
    'friend': {
        'name' : 'Robert'
    }
}
absolute refs: 

{ '$ref': 'http://data.com/person/#'}

type discriminator in gform schema

{
'type' : 'multi-object',
'typeProperty': '__type',
'groups': [
    {
        'code':'Car',
        'attributes':[...]
    },{
        'code':'Truck',
        'attributes':[...]
    }
]}


{
'__type': 'Car',
'name': 'Dudu'
}
          

{
    '__type': 'Truck',
    'size': '5000'
}
